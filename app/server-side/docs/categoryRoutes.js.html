const EXPRESS = require('express');
const ROUTER = EXPRESS.Router();
const CATEGORY = require('../models/category');

/**
 * @module category
 * @description Маршруты для работы с категориями
 */

/**
 * Получает список всех категорий из базы данных
 * @name getCategories
 * @route GET /category
//  * @memberof module:category
 * @returns {Object} - Объект ответа
 * @returns {module:models/category.Category[]} categoryArray - Массив категорий
 * 
 * @example
 * // Ответ:
 * {
 *   "categoryArray": [
 *     {
 *       "name": "Example",
 *       "translateOne": "пример",
 *       "translateMany": "примеры",
 *       "filters": [...]
 *     }
 *   ]
 * }
 */
ROUTER.get('/', async (request, result) => {
    try {
        const CATEGORY_ARRAY = await CATEGORY.find();
        result.status(201).json({ categoryArray: CATEGORY_ARRAY });
    } catch (error) {
        result.status(400).json({ message: error.message });
    }
});

module.exports = ROUTER;
