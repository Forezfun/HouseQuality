const MONGOOSE = require('mongoose');

/**
 * @typedef {Object} ImageFurniture
 * @memberof module:furnitureCard
 * @property {string} filename - Имя изображения, сохранённое на сервере.
 */

/**
 * @typedef {Object} ImagesFurniture
 * @memberof module:furnitureCard
 * @property {string} color - Цвет, к которому относятся изображения.
 * @property {module:furnitureCard.ImageFurniture[]} images - Список изображений.
 * @property {number} idMainImage - Индекс главного изображения в массиве `images`.
 * @property {string} furnitureCardId - ID карточки мебели, к которой относятся изображения.
 */

const IMAGES_FURNITURE_SCHEM = new MONGOOSE.Schema({
    color: String,
    images: [{
        filename: String
    }],
    idMainImage: Number,
    furnitureCardId: String
});

const IMAGES_FURNITURE = MONGOOSE.model('ImagesFurniture', IMAGES_FURNITURE_SCHEM);

module.exports = IMAGES_FURNITURE;
